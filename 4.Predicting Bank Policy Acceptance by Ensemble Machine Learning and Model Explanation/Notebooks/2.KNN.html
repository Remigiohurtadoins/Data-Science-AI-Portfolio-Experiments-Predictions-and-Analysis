# Funcin para medir la calidad de modelos
def obtenerResultados(y_test, y_pred):
    accuracy = accuracy_score(y_test, y_pred)
    precision = precision_score(y_test, y_pred, average='macro')
    recall = recall_score(y_test, y_pred, average='macro')
    f1 = f1_score(y_test, y_pred, average='macro')
    
    accuracy=str(round(accuracy, 4))
    precision=str(round(precision, 4))
    recall=str(round(recall, 4))
    f1=str(round(f1, 4))
    
    print("Accuracy:", accuracy)
    print("Precision:", precision)
    print("Recall o Sensitivity:", recall)
    print("F1-Score:", f1)
    return accuracy, precision, recall, f1

def train_test_metrics(pipe, X_train, X_test, y_train, y_test):
    # make predictions for train set
    print('Resultados en Train:')
    y_pred = pipe.predict(X_train)
    obtenerResultados(y_train, y_pred)

    # make predictions for test set
    print('\nResultados en Test:')
    y_pred = pipe.predict(X_test)
    obtenerResultados(y_test, y_pred)
    print("Confusion Matrix:\n", confusion_matrix(y_test, y_pred))

from sklearn.ensemble import RandomForestClassifier, StackingClassifier
from sklearn.model_selection import train_test_split, GridSearchCV
from sklearn.metrics import accuracy_score, classification_report, confusion_matrix
# Crear modelos base
knn = KNeighborsClassifier(n_neighbors=5)
rf = RandomForestClassifier(n_estimators=200, max_features='log2', max_depth=30, criterion='entropy', random_state=0)

# Crear un StackingClassifier
estimators = [
    ('knn', knn),
    ('rf', rf)
]

stacking_clf = StackingClassifier(estimators=estimators, final_estimator=RandomForestClassifier())

# Ajustar hiperparmetros usando GridSearchCV
param_grid = {
    'knn__n_neighbors': [3, 5, 7, 10, 15, 20],
    'rf__n_estimators': [50, 100, 150],
    'rf__max_depth': [10, 20, 30, 50]
}

grid_search = GridSearchCV(stacking_clf, param_grid, cv=3, scoring='accuracy', n_jobs=-1)
grid_search.fit(X_train_scaled, y_train)

# Evaluar el modelo
y_pred = grid_search.best_estimator_.predict(X_test_scaled)
print("Evaluacin del modelo StackingClassifier:")
print("Classification Report:\n", classification_report(y_test, y_pred))

train_test_metrics(grid_search.best_estimator_, X_train_scaled, X_test_scaled, y_train, y_test)

best_params = grid_search.best_params_
print("Best parameters: ", best_params)

# Identificar las variables ms relevantes a partir del modelo Random Forest en el StackingClassifier

c = cv_results['mean_test_score']

# Obtener valores nicos para cada parmetro
x_unique = np.unique(x)
y_unique = np.unique(y)
z_unique = np.unique(z)

# Crear una malla de los parmetros
X, Y, Z = np.meshgrid(x_unique, y_unique, z_unique, indexing='ij')

# Inicializar la matriz de puntuaciones
mean_scores = np.full(X.shape, np.nan)

# Llenar la matriz de puntuaciones con los valores de mean_test_score
for xi in range(X.shape[0]):
    for yi in range(Y.shape[1]):
        for zi in range(Z.shape[2]):
            mask = (x == X[xi, yi, zi]) & (y == Y[xi, yi, zi]) & (z == Z[xi, yi, zi])
            if np.any(mask):
                mean_scores[xi, yi, zi] = np.mean(c[mask])

# Crear la figura
fig = plt.figure(figsize=(12, 8))
ax = fig.add_subplot(111, projection='3d')
